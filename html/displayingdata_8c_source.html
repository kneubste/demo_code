284/** **************************************************************************
285 * @brief Function for displaying the angle data
286 * @note This function calls the required functions to display the data for an angle measurement.
287 * @n Shows 2 dots on the screen for visualization of the direction
288 * @n Shows an error on display if the data is unclear
289 * @n Clears the ADC_samples array after displaying all the data
290 *****************************************************************************/
291
292void DISP_show_data_angle(void)
293{
294 const uint32_t X_SIZE = 240;
295 const uint32_t Y_OFFSET_PAD = 280;
296 char text[16];
297
298 int32_t angle = 666;
299 angle = angle_to_cable();
300
301 /* Clear the display */
302 BSP_LCD_SetTextColor(LCD_COLOR_WHITE);
303 BSP_LCD_FillRect(0, 0, X_SIZE, Y_OFFSET_PAD+1);
304 /* Write the measurement of the angle */
305 BSP_LCD_SetFont(&Font24);
306 BSP_LCD_SetBackColor(LCD_COLOR_WHITE);
307 BSP_LCD_SetTextColor(LCD_COLOR_BLACK);
308 BSP_LCD_DisplayStringAt(5,10, (uint8_t *)"Angle", LEFT_MODE);
309 BSP_LCD_SetFont(&Font20);
310 BSP_LCD_DisplayStringAt(5,50, (uint8_t *)"Value in Degree", LEFT_MODE);
311
312 //Two points for a rough indication of direction
313 BSP_LCD_DrawCircle(45,220,20);
314 BSP_LCD_DrawCircle(195,220,20);
315
316 //Color middle part for direction / error
317 if(CALC_degree_left){
318 CALC_degree_left=false;
319 snprintf(text, 15, "Angle: %4d", (int)(angle));
320 BSP_LCD_DisplayStringAt(5, 90, (uint8_t *)text, LEFT_MODE);
321 BSP_LCD_SetTextColor(LCD_COLOR_GREEN);
322 BSP_LCD_FillCircle(45,220,10);
323 }
324 else if(CALC_degree_right){
325 CALC_degree_right=false;
326 snprintf(text, 15, "Angle: %4d", (int)(angle));
327 BSP_LCD_DisplayStringAt(5, 90, (uint8_t *)text, LEFT_MODE);
328 BSP_LCD_SetTextColor(LCD_COLOR_GREEN);
329 BSP_LCD_FillCircle(195,220,10);
330 }
331 else{
332 BSP_LCD_SetTextColor(LCD_COLOR_RED);
333 BSP_LCD_DisplayStringAt(5,90, (uint8_t *)"NO VALUE", CENTER_MODE);
334 BSP_LCD_FillCircle(120,150,10);
335 BSP_LCD_FillCircle(45,220,10);
336 BSP_LCD_FillCircle(195,220,10);
337 }
