65/** **************************************************************************
66 * @brief Calculate the average of the ADC samples
67 * @param numb_samples Number of samples
68 * @param arr[] Array filed with samples
69 * @note -
70 * @return Calculated average of the ADC samples
71 *****************************************************************************/
72int32_t average(int32_t numb_samples, int32_t arr[]){
73 int i;
74 int32_t avg = 0;
75
76 for(i=0; i<numb_samples; i++){
77 avg += arr[i];
78 }
79 avg /= numb_samples;
80
81 return avg;
82}

84/** **************************************************************************
85 * @brief calculate the RMS value
86 * @param numb_samples: number of samples
87 * @param arr[]: array filled with samples
88 * @note -
89 * @return calculated RMS value
90 *****************************************************************************/
91int32_t RMS(int32_t numb_samples, int32_t arr[]){
92 int i;
93 int32_t avg = 0;
94 int32_t rms = 0;
95 int32_t tmp = 0;
96 int32_t diff = 0;
97
98 avg = average(numb_samples, arr);
99
100 for(i=0; i<numb_samples; i++){
101 diff = arr[i] - avg;
102 tmp = diff*diff;
103 rms = rms + tmp;
104 }
105 rms = rms/numb_samples;
106 rms = sqrt(rms);
107
108 return rms;
109}

114/** **************************************************************************
115 * @brief find the distance between device and cable
116 * @note measure in the range of [5,200]mm and has a precision of -/+30%
117 * @param meas_mode 1 = single measurement (10 samples),
118 * else = accurate measurement (50 samples)
119 * @param meas_type 1 = wire

121 * @return distance to the cable up to 200mm
122 *****************************************************************************/
123int32_t distance_to_cable(int32_t meas_mode){
124

126 int32_t e_val=0; //< e_val is the electrical field value
127
128
129

131 pad1 = RMS(10, PAD1_samples);

149 }
150 else{
151 while(lut_pad_wire[dist]>e_val){
152 dist++;
153 }
154 return dist;}
155 }
156 else if(MEAS_data_cable){

158 dist=0;

164 }

177/** **************************************************************************
178 * @brief calculate the angle to the cable
179 * @note range [-45,45]°, precision -/+15
180 * padleft:@0mm=1090,@20mm=277
181 * padright:@0mm=1060,@20mm=326
182 * pad1=pad2=764
183 * pad2-pad1=783
184 * @n The field is very location and board depending.
185 * @return angle to wire/cable
186 *****************************************************************************/
187int32_t angle_to_cable(){
188
189 float diff;
190 float ratio;
191 float angle;
192 int32_t pad1 = 0;
193 int32_t pad2 = 0;
194
195 pad1 = RMS(50, PAD1_samples);
196 pad2 = RMS(50, PAD2_samples);
197
198 diff = pad1 - pad2;
199
200 if(diff > 232){
201 diff = 232;
202 }
203
204 if(diff < -222){
205 diff=-222;
206 }
207
208 if(diff > 20)
209 {
210 ratio = diff/232;
211 angle= 45*ratio;
212 CALC_degree_left=true;
213 }
214 else
215 {
216 ratio = diff/222;
217 angle= 45*ratio;
218 CALC_degree_right=true;
219 }
220
221
222
223 return angle;
224
225}
